\section{Introduction}
JLPT N5 is the lowest level of the Japanese-Language Proficiency Test. These
notes cover the core hiragana, katakana, basic vocabulary, and grammar needed
to reach this level.

\section{Hiragana}
% Example of typesetting Japanese: あ い う え お

\subsection{Typing small kana (mozc)}
Small kana change the sound and must be typed deliberately.

\textbf{Combined sounds} --- type the whole syllable and the small ゃ/ゅ/ょ
appears automatically:
\begin{center}
\begin{tabular}{@{}ll@{}}
\toprule
Type & Result \\
\midrule
\texttt{sho} / \texttt{syo} & しょ \\
\texttt{kyo} & きょ \\
\texttt{cho} & ちょ \\
\bottomrule
\end{tabular}
\end{center}

\textbf{Small kana on their own} --- prefix with \texttt{x} (or \texttt{l}):
\begin{center}
\begin{tabular}{@{}ll@{}}
\toprule
Type & Result \\
\midrule
\texttt{xtu} & っ (small tsu --- also made by doubling a consonant, e.g.\ \texttt{kakkoii}) \\
\texttt{xya} & ゃ \\
\texttt{xyu} & ゅ \\
\texttt{xyo} & ょ \\
\texttt{xa} & ぁ \\
\bottomrule
\end{tabular}
\end{center}
Rule of thumb: \texttt{x} (or \texttt{l}) before a kana gives its small version.

\section{Katakana}
Katakana is used mainly for \textbf{loanwords} (words borrowed from other
languages). The long-vowel mark \textbf{ー} (\emph{chōonpu}) marks a lengthened
vowel.
\begin{center}
\begin{tabular}{@{}llll@{}}
\toprule
Katakana & Romaji & Origin & English \\
\midrule
ノート   & nōto   & English ``note''       & notebook \\
コーヒー & kōhī   & English ``coffee''     & coffee \\
テレビ   & terebi & English ``television'' & TV \\
\bottomrule
\end{tabular}
\end{center}
mozc tip: type the word (e.g.\ \texttt{no-to}, where \texttt{-} gives ー) and
press \texttt{F7} to convert to katakana.

\subsection{Adjectives}
\begin{center}
\begin{tabular}{@{}llll@{}}
\toprule
Japanese & Romaji & Type (mozc) & English \\
\midrule
いい     & ii      & i i        & good \\
かっこいい & kakkoii & ka k ko i i & cool / stylish \\

\bottomrule
\end{tabular}

Note the small っ in か\textbf{っ}こいい --- type the doubled consonant

\textbf{い-adjectives vs な-adjectives:} いい and かっこいい are
\emph{い-adjectives} (they end in い). 完璧 (かんぺき) is a \emph{な-adjective}:
it takes な before a noun --- 完璧\textbf{な}質問 (``a perfect question'') --- and
完璧です to end a sentence.

\section{Grammar}

\subsection{Particle \texorpdfstring{と}{to} (``with'' / ``and'')}
The particle \textbf{と} has two common N5 uses:
\begin{itemize}
  \item ``\textbf{with}'' (a companion): 彼女\textbf{と}日本語を勉強します。 ---
        ``I study Japanese \emph{with} my girlfriend.''
  \item ``\textbf{and}'' (listing nouns): パン\textbf{と}コーヒー --- ``bread
        \emph{and} coffee.''
\end{itemize}
It goes \emph{after} the noun it attaches to (彼女と = ``with her'').

\subsection{Existence: \texorpdfstring{あります}{arimasu} / \texorpdfstring{います}{imasu}}
To say something exists (or someone has something), use the pattern:
\begin{center}
  \textbf{X に　Y が　あります} \quad(\emph{X ni Y ga arimasu})
\end{center}
where \textbf{に} marks the location (where Y is) and \textbf{が} marks the
thing that exists.

\begin{itemize}
  \item あります --- used for \emph{inanimate} things (objects, words, places).
  \item います --- used for \emph{living} things (people, animals).
\end{itemize}

Example:
\begin{quote}
ありがとうに「う」があります。
\end{quote}
\emph{Arigatou ni ``u'' ga arimasu.} --- ``There is an `う' in ありがとう.''

\textbf{Negative and other forms of あります:}
\begin{center}
\begin{tabular}{@{}lll@{}}
\toprule
Form & Japanese & Meaning \\
\midrule
affirmative   & あります             & there is / I have \\
negative      & ありません           & there isn't / I don't have \\
past          & ありました           & there was \\
past negative & ありませんでした     & there wasn't \\
\bottomrule
\end{tabular}
\end{center}
\begin{quote}
時間がありません。 --- ``(I/we) don't have time.'' (lit.\ ``time does not exist'')
\end{quote}

\textbf{General rule:} polite \textbf{〜ます $\rightarrow$ 〜ません} makes any verb
negative --- 行きます $\rightarrow$ 行きません (``don't go''),
わかります $\rightarrow$ わかりません (``don't understand'').

\subsection{Indefinites: question word + \texorpdfstring{か}{ka}}
Adding \textbf{か} to a question word makes it indefinite (``some-/any-''). This
か is \emph{not} the question particle.
\begin{center}
\begin{tabular}{@{}llll@{}}
\toprule
Question word & + か & Romaji & Meaning \\
\midrule
何 (what)   & 何か   & nanika & something / anything \\
誰 (who)    & 誰か   & dareka & someone / anyone \\
どこ (where)& どこか & dokoka & somewhere / anywhere \\
いつ (when) & いつか & itsuka & sometime / someday \\
\bottomrule
\end{tabular}
\end{center}
Example:
\begin{quote}
何か他にありますか。 --- ``Is there anything else?''
\end{quote}
Here there are \emph{two} か: 何\textbf{か} (anything) + the final \textbf{か}
(question particle).

\textbf{か vs が --- different meanings:}
\begin{itemize}
  \item 何\textbf{か}ありますか。--- ``Is there \emph{anything}?'' (yes/no)
  \item 何\textbf{が}ありますか。--- ``\emph{What} is there?'' (asks you to name things)
\end{itemize}

\subsection{Question word: \texorpdfstring{どうやって}{dou yatte}}
\textbf{どうやって} (\emph{dou yatte}) means ``how'' in the sense of
\emph{by what means / method}. It breaks down as:
\begin{itemize}
  \item どう (\emph{dou}) --- how / in what way
  \item やって (\emph{yatte}) --- て-form of やる (\emph{yaru}, ``to do'')
\end{itemize}
Literally ``doing in what way'' $\rightarrow$ ``how / by what means.''

Use it when the answer is a \emph{method} (walking, by train, mixing
ingredients, etc.):
\begin{quote}
駅までどうやって行きますか。 --- ``How do I get to the station?''
\end{quote}

Contrast with plain \textbf{どう}, which asks about a state or impression:
\begin{quote}
どうですか。 --- ``How is it? / How are you?''
\end{quote}

\subsection{Forms of \texorpdfstring{やる}{yaru} (``to do'')}
A single verb branches into several forms. For やる (\emph{yaru}):
\begin{center}
\begin{tabular}{@{}lll@{}}
\toprule
Form & Japanese & Use \\
\midrule
Dictionary  & やる   & ``to do'' \\
て-form      & やって & connector --- ``doing / and do'' (as in どうやって) \\
Casual past & やった & ``did'' --- also the cheer 「やった！」 = ``I did it! / Yay!'' \\
\bottomrule
\end{tabular}
\end{center}
Note that やって and やった differ only in the last kana (て vs た) and both
use the small っ (typed \texttt{yatte} / \texttt{yatta}).

\subsection{Sentence-final particles: \texorpdfstring{よ}{yo} / \texorpdfstring{ね}{ne}}
These attach to the end of a sentence to add nuance:
\begin{itemize}
  \item \textbf{よ} (\emph{yo}) --- emphasis; \emph{informing/asserting} something
        to the listener (``\ldots you know!'').
  \item \textbf{ね} (\emph{ne}) --- seeks \emph{agreement} (``\ldots right?'').
\end{itemize}
\begin{center}
\begin{tabular}{@{}ll@{}}
\toprule
Sentence & Feel \\
\midrule
かっこいいです。   & ``It's cool.'' (neutral) \\
かっこいいです\textbf{よ}。 & ``It's cool, you know!'' (asserting) \\
かっこいいです\textbf{ね}。 & ``It's cool, isn't it?'' (seeking agreement) \\
\bottomrule
\end{tabular}
\end{center}

\subsection{Homophones: \texorpdfstring{起きる}{okiru} vs \texorpdfstring{置く}{oku} (\texorpdfstring{おきます}{okimasu})}
Both verbs become \textbf{おきます} in the polite form --- in kana alone they
are identical, and only the kanji (or context) tells them apart:
\begin{center}
\begin{tabular}{@{}llll@{}}
\toprule
Kanji & Verb & Meaning & Object? \\
\midrule
\textbf{起}きます & 起きる (\emph{okiru}) & to wake up / get up & no (intransitive) \\
\textbf{置}きます & 置く (\emph{oku})    & to put / place      & yes, takes を (transitive) \\
\bottomrule
\end{tabular}
\end{center}
\textbf{Disambiguation trick:} if there is a direct object marked with
\textbf{を}, it is 置く (put). If there is no を (just a time/place), it is
起きる (wake up).
\begin{itemize}
  \item 朝6時に\textbf{起きます}。 --- ``I get up at 6 a.m.'' (no を)
  \item かばん\textbf{を}ここに\textbf{置きます}。 --- ``I put the bag here.'' (を)
\end{itemize}
This is a clear example of why kanji matters: it instantly distinguishes words
that sound identical.

\subsection{Irregular past of \texorpdfstring{いい}{ii} (``good'')}
\textbf{いい} (good) has an \emph{irregular} conjugation: its stem secretly
changes to よ-.
\begin{center}
\begin{tabular}{@{}lll@{}}
\toprule
Tense & Japanese & Meaning \\
\midrule
present & いい       & (it's) good \\
past    & よかった   & (it) was good / I'm glad \\
\bottomrule
\end{tabular}
\end{center}
Never say いかった ($\times$) --- it is always \textbf{よかった}. Used a lot as
an emotional reaction: 「よかった！」= ``Oh good! / I'm so glad!''

\textbf{Polite past of い-adjectives:} add です to the plain past --- do
\emph{not} also change です.
\begin{itemize}
  \item ($\checkmark$) よかった\textbf{です}。 --- ``It was good.''
  \item ($\times$) よかったでした。 --- common mistake; never double the past.
\end{itemize}

\subsection{Demonstratives (\texorpdfstring{こそあど}{kosoado}): \texorpdfstring{これ}{kore} vs \texorpdfstring{この}{kono}}
\begin{center}
\begin{tabular}{@{}llll@{}}
\toprule
 & Near me (こ) & Near you (そ) & Over there (あ) \\
\midrule
Pronoun (``this/that \textbf{one}'', stands alone) & これ & それ & あれ \\
+ noun (``this/that \textbf{<noun>}'')             & この & その & あの \\
\bottomrule
\end{tabular}
\end{center}
\textbf{Key:} これ stands alone; この must be followed by a noun.
\begin{itemize}
  \item これはよかったです。 --- ``This one was good.'' (これ alone)
  \item この単語はよかったです。 --- ``This word was good.'' (この + noun 単語)
\end{itemize}

\subsection{\texorpdfstring{知っています}{shitte imasu} (to know) --- class example}
To ask whether someone knows a person (or thing), use:
\begin{center}
  \textbf{[Person]を知っていますか。} \quad(\emph{[Person] wo shitte imasu ka})
\end{center}
``Do you know [Person]?'' --- e.g.\ 田中さんを知っていますか。

``To know'' is expressed as a \emph{state} using the て-form + います
(知る $\rightarrow$ 知\textbf{って}, small っ):
\begin{center}
\begin{tabular}{@{}lll@{}}
\toprule
Meaning & Japanese & Note \\
\midrule
Do you know? & 知っていますか & て-form + います \\
I know       & 知っています   & \\
I don't know & 知りません     & irregular --- \emph{not} 知っていません \\
\bottomrule
\end{tabular}
\end{center}
Note the irregular negative: ``I don't know'' is \textbf{知りません}
(\emph{shirimasen}), not 知っていません.

\section{Phrases}

\subsection{Self-introduction}
\begin{quote}
こんにちは。私はティオです。どうぞよろしくお願いします。
\end{quote}
\emph{Konnichiwa. Watashi wa Tio desu. Douzo yoroshiku onegai shimasu.}\\
``Hello. I am Theoden. Pleased to meet you.''

\begin{itemize}
  \item 私 (\emph{watashi}) --- I
  \item は (\emph{wa}) --- topic particle (typed \texttt{ha})
  \item です (\emph{desu}) --- polite copula, ``am/is''
  \item どうぞよろしくお願いします --- set phrase, ``pleased to meet you''
\end{itemize}

\subsection{Studying phrases}
\begin{center}
\begin{tabular}{@{}lll@{}}

\section{駅 (At the Station)}

\subsection{Directions}

\textbf{Asking the way:}
\begin{quote}
駅までどうやって行きますか。
\end{quote}
\emph{Eki made dou yatte ikimasu ka.} --- ``How do I get to the station?''

\begin{itemize}
  \item 駅 (\emph{eki}) --- (train) station
  \item まで (\emph{made}) --- up to / as far as (destination)
  \item どうやって (\emph{dou yatte}) --- how / by what means
  \item 行きます (\emph{ikimasu}) --- go (ます-form)
  \item か --- question particle
\end{itemize}

\textbf{Answering (with 歩く, ``to walk''):}
\begin{quote}
歩いて行きます。\quad / \quad 駅まで歩きます。
\end{quote}
\emph{Aruite ikimasu. / Eki made arukimasu.} --- ``I go on foot. / I walk to
the station.''

Conjugations of 歩く (\emph{aruku}): ます-form \textbf{歩きます}
(\emph{arukimasu}); て-form \textbf{歩いて} (\emph{aruite}). Note how
\textbf{歩いて行きます} uses the て-form to connect 歩く to 行く --- the て-form
links verbs, while \textbf{ます} ends the sentence.


\section{Conversation}

\subsection{\texorpdfstring{あいづち}{aizuchi} (backchanneling)}
The most important listening behaviour in Japanese is \textbf{あいづち}
(\emph{aizuchi}) --- frequent short signals that show ``I'm listening, keep
going.'' They are used far more often than in English; staying silent can seem
inattentive or like disagreement.
\begin{center}
\begin{tabular}{@{}lll@{}}
\toprule
Japanese & Romaji & Feel \\
\midrule
はい / ええ / うん & hai / ee / un & ``yes / mm-hm'' (formal $\rightarrow$ casual) \\
そうですね         & sou desu ne   & ``that's right / I see'' \\
そうですか         & sou desu ka   & ``oh, is that so?'' \\
なるほど           & naruhodo      & ``I see / makes sense'' \\
へえ〜             & hee           & ``wow / really!'' (surprise) \\
\bottomrule
\end{tabular}
\end{center}

\subsection{\texorpdfstring{オウム返し}{oumu-gaeshi} (echoing)}
\textbf{オウム返し} (literally ``parrot return'') is repeating a \emph{key word}
back --- often with rising intonation --- to show you are following, confirm
what you heard, or express surprise. Use a light touch (a word or two), not full
sentences.
\begin{quote}
A: 昨日、京都に行きました。 --- ``Yesterday I went to Kyoto.''\\
B: 京都に！いいですね。 --- ``To Kyoto! Nice!''
\end{quote}
Think あいづち first, オウム返し second: the goal is shared rhythm and presence,
not summarising everything back.
